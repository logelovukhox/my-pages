欢迎来到前端世界

这是一个简单的响应式前端页面示例，展示了HTML、CSS和JavaScript的基本用法。

example.js 
// 简单的JavaScript示例
function greetUser(name) {
    return `你好, ${name}!`;
}

// 事件监听器
document.addEventListener('DOMContentLoaded', () => {
    const button = document.getElementById('demoBtn');
    button.addEventListener('click', () => {
        alert(greetUser('访客'));
    });
});
